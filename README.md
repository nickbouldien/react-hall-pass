# react-hall-pass

>

## Install

with yarn:
```bash
yarn add react-hall-pass
```

or without yarn:
```bash
npm install --save react-hall-pass
```

## Usage

```tsx
import * as React from "react";

import HallPass from "react-hall-pass";

class Example extends React.Component {
  render() {
    return <HallPass />;
  }
}
```
